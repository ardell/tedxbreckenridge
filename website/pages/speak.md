<div class="mb-80">
  <h2 class="mb-32">Our Selection Process</h2>

  <div class="grid-gap-32">
    <div class="card-accent-red">
      <h3>Live Auditions</h3>
      <p>20 applicants selected for 3-minute live auditions in Breckenridge</p>
    </div>

    <div class="card-accent-teal">
      <h3>Main Stage Selection</h3>
      <p>6-8 speakers and 2 performers invited to main stage</p>
    </div>

    <div class="card-accent-red">
      <h3>Speaker Coaching</h3>
      <p>Free required speaker coaching provided with one-on-one coach pairing for talk development</p>
    </div>
  </div>
</div>

<div class="mb-80">
  <h2 class="mb-32">What NOT to Submit</h2>

  <p class="mb-24">Talks should avoid being:</p>

  <div class="checklist-x mb-32">
    <div class="checklist-x-item">
      <p>Motivational speeches with broad themes</p>
    </div>
    <div class="checklist-x-item">
      <p>Personal stories without drawn ideas/lessons</p>
    </div>
    <div class="checklist-x-item">
      <p>Lectures or biographies</p>
    </div>
    <div class="checklist-x-item">
      <p>Recycled professional circuit material</p>
    </div>
    <div class="checklist-x-item">
      <p>Lacking a single central idea</p>
    </div>
  </div>

  <div class="callout-gradient">
    <p>"Great TEDx Talks center on one single Idea Worth Spreading."</p>
  </div>
</div>

<div class="mb-80">
  <h2 class="mb-32">Additional Requirements</h2>

  <div class="checklist">
    <div class="checklist-item">
      <p>All cited data/facts must be fact-checked by applicant</p>
    </div>
    <div class="checklist-item">
      <p>Credible data and references required</p>
    </div>
  </div>
</div>
